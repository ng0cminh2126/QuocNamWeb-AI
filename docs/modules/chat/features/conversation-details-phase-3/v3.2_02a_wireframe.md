## âŒ¨ï¸ Keyboard Shortcuts

| Key                | Action        | Availability    |
| ------------------ | ------------- | --------------- |
| ESC                | Close modal   | Always          |
| Arrow Right / Down | Next page     | Multi-page only |
| Arrow Left / Up    | Previous page | Multi-page only |
| Home               | First page    | Multi-page only |
| End                | Last page     | Multi-page only |
| Enter              | N/A           | -               |

**Implementation:**

```tsx
useEffect(() => {
  const handleKeyDown = (e: KeyboardEvent) => {
    if (e.key === "Escape") closeModal();
    if (totalPages === 1) return; // No navigation for single-page

    if (e.key === "ArrowRight" || e.key === "ArrowDown") {
      if (currentPage < totalPages) nextPage();
    }
    if (e.key === "ArrowLeft" || e.key === "ArrowUp") {
      if (currentPage > 1) prevPage();
    }
    if (e.key === "Home") goToPage(1);
    if (e.key === "End") goToPage(totalPages);
  };

  document.addEventListener("keydown", handleKeyDown);
  return () => document.removeEventListener("keydown", handleKeyDown);
}, [currentPage, totalPages]);
```
